275 auto& historian = *(work.historian);
276 auto it = historian.cache_exists.find(i);
277 if (it != historian.cache_exists.end()) {
278 auto chosen = it->second;
279 historian.cache_data.splice(historian.cache_data.end(), historian.cache_data, chosen); // move to end.
280 return Extracted(*chosen);

283 // Check if bounds already exist from the reusable cache. If so,
284 // we can use them to reduce the amount of data I/O.
285 hsize_t extraction_start = pointers[i];
286 hsize_t extraction_len = pointers[i + 1] - pointers[i];
287 bool bounded = false;
288
289 if (work.extraction_bounds.size()) {
290 const auto& current = work.extraction_bounds[i];
291 if (current.first != -1) {
292 bounded = true;
293 extraction_start = current.first;
294 extraction_len = current.second;

298 // Need to use the max_cache_number as the capacity of each recycled
299 // vector may be much larger than the reported size of the chunk; this
300 // would cause us to overrun the cache_size_limit if we recycled the
301 // vector enough times such that each cache element had capacity equal
302 // to the maximum number of non-zero elements in any dimension element.
303 //
304 // Alternatives would be to create a new Element on every recycling
305 // iteration, or to hope that shrink_to_fit() behaves. Both would allow
306 // us to store more cache elements but would involve reallocations,
307 // which degrades perf in the most common case where a dimension
308 // element is accessed no more than once during iteration.
309 if (historian.max_cache_number == -1) {
310 historian.max_cache_number = cache_size_limit / (max_non_zeros * ((needs_value ? sizeof(Value_) : 0) + sizeof(Index_)));
311 if (historian.max_cache_number == 0) {
312 historian.max_cache_number = 1;
313 }
314 }
315
316 // Adding a new last element, or recycling the front to the back.
317 // We initialize each element with the maximum number of non-zeros to avoid reallocations.
318 typename std::list<typename LruCache::Element>::iterator location;
319 if (historian.cache_data.size() < historian.max_cache_number) {
320 historian.cache_data.push_back(typename LruCache::Element());
321 location = std::prev(historian.cache_data.end());
322 location->index.resize(max_non_zeros);
323 if (needs_value) {
324 location->value.resize(max_non_zeros);
325 }
326 } else {
327 location = historian.cache_data.begin();
328 historian.cache_exists.erase(location->id);
329 historian.cache_data.splice(historian.cache_data.end(), historian.cache_data, location); // move to end.
330 }
331 historian.cache_exists[i] = location;
332
333 auto& current_cache = *location;
334 current_cache.id = i;
335 current_cache.length = extraction_len;
336 current_cache.bounded = bounded;

364 static void sort_by_field(std::vector<int>& indices, Function_ field) {
365 auto comp = [&field](size_t l, size_t r) -> bool {
366 return field(l) < field(r);
367 };
368
369 bool sorted = true;
370 for (size_t i = 1, end = indices.size(); i < end; ++i) {
371 if (!comp(indices[i-1], indices[i])) {
372 sorted = false;
373 break;
374 }
375 }
376
377 if (!sorted) {
378 std::sort(indices.begin(), indices.end(), comp);
379 }

382 Extracted extract_primary_with_oracle(PrimaryWorkspace& work, bool needs_value) const {
383 auto& pred = *work.futurist;
384 if (pred.predictions_made.size() > pred.predictions_fulfilled) {
385 auto chosen = pred.predictions_made[pred.predictions_fulfilled++];
386 return Extracted(pred, chosen, needs_value);
387 }
388
389 // Grow the number of predictions over time, until we get to a point
390 // where we consistently fill the cache.
391 size_t max_predictions = pred.predictions_made.size() * 2;
392 if (max_predictions < 100) {
393 max_predictions = 100;
394 } else {
395 size_t upper = (row_ ? nrows : ncols);
396 if (max_predictions > upper) {
397 max_predictions = upper;
398 }
399 }
400
401 pred.predictions_made.clear();
402 pred.needed.clear();
403 pred.present.clear();
404 pred.next_cache_data.clear();
405 pred.next_cache_exists.clear();
406
407 // Here, we use a giant contiguous buffer to optimize for
408 // near-consecutive iteration. This allows the HDF5 library to pull out
409 // long strips of data from the file. It also allows us to maximize
410 // the use of the cache_size_limit by accounting for differences in the
411 // non-zeros for each element, rather than conservatively assuming
412 // they're all at max (as in the LRU case). The downside is that we
413 // need to do some copying within the cache to make space for new
414 // reads, but that works out to be no more than one extra copy per
415 // fetch() call, which is tolerable. I suppose we could do better
416 // by defragmenting within this buffer but that's probably overkill.
417 if (pred.max_cache_elements == -1) {
418 pred.max_cache_elements = cache_size_limit / ((needs_value ? sizeof(Value_) : 0) + sizeof(Index_));
419 if (pred.max_cache_elements < max_non_zeros) {
420 pred.max_cache_elements = max_non_zeros; // make sure we have enough space to store the largest possible primary dimension element.
421 }
422 pred.cache_index.resize(pred.max_cache_elements);
423 if (needs_value) {
424 pred.cache_value.resize(pred.max_cache_elements);
425 }
426 }
427 size_t filled_elements = 0;
428
429 for (size_t p = 0; p < max_predictions; ++p) {
430 Index_ current;
431 if (!pred.prediction_stream.next(current)) {
432 break;
433 }
434
435 // Seeing if this element already exists somewhere.
436 auto nit = pred.next_cache_exists.find(current);
437 if (nit != pred.next_cache_exists.end()) {
438 pred.predictions_made.push_back(nit->second);
439 continue;
440 }
441
442 auto it = pred.cache_exists.find(current);
443 if (it != pred.cache_exists.end()) {
444 auto& candidate = pred.cache_data[it->second];
445 filled_elements += candidate.length;
446 if (filled_elements > pred.max_cache_elements) {
447 pred.prediction_stream.back();
448 break;
449 }
450
451 Index_ used = pred.next_cache_data.size();
452 pred.predictions_made.push_back(used);
453 pred.present.push_back(used);
454 pred.next_cache_exists[current] = used;
455 pred.next_cache_data.push_back(std::move(candidate));
456 continue;
457 }
458
459 // Check if bounds already exist from the reusable cache. If so,
460 // we can use them to reduce the amount of data I/O.
461 hsize_t extraction_start = pointers[current];
462 hsize_t extraction_len = pointers[current + 1] - pointers[current];
463 bool bounded = false;
464
465 if (work.extraction_bounds.size()) {
466 const auto& bounds = work.extraction_bounds[current];
467 if (bounds.first != -1) {
468 bounded = true;
469 extraction_start = bounds.first;
470 extraction_len = bounds.second;
471 }
472 }
473
474 filled_elements += extraction_len;
475 if (filled_elements > pred.max_cache_elements) {
476 pred.prediction_stream.back();
477 break;
478 }
479
480 Index_ used = pred.next_cache_data.size();
481 pred.predictions_made.push_back(used);
482 pred.needed.emplace_back(used);
483 pred.next_cache_exists[current] = used;
484
485 typename OracleCache::Element latest;
486 latest.data_offset = extraction_start;
487 latest.length = extraction_len;
488 latest.bounded = bounded;
489 pred.next_cache_data.push_back(std::move(latest));
490 }
491
492 if (pred.needed.size()) {
493 size_t dest_offset = 0;
494
495 if (pred.present.size()) {
496 // Shuffling all re-used elements to the start of the buffer,
497 // so that we can perform a contiguous extraction of the needed
498 // elements in the rest of the buffer. This needs some sorting
499 // to ensure that we're not clobbering one re-used element's
500 // contents when shifting another element to the start.
501 sort_by_field(pred.present, [&pred](size_t i) -> size_t { return pred.next_cache_data[i].mem_offset; });
502
503 for (const auto& p : pred.present) {
504 auto& info = pred.next_cache_data[p];
505 auto isrc = pred.cache_index.begin() + info.mem_offset;
506
507#ifdef DEBUG
508 if (info.mem_offset < dest_offset) {
509 throw std::runtime_error("detected clobbering of memory cache from overlapping offsets");
510 }
511#endif
512
513 std::copy(isrc, isrc + info.length, pred.cache_index.begin() + dest_offset);
514 if (needs_value) {
515 auto vsrc = pred.cache_value.begin() + info.mem_offset;
516 std::copy(vsrc, vsrc + info.length, pred.cache_value.begin() + dest_offset);
517 }
518 info.mem_offset = dest_offset;
519 dest_offset += info.length;
520 }
521 }
522
523 // Sorting so that we get consecutive accesses in the hyperslab construction.
524 // This should improve re-use of partially read chunks inside the HDF5 call.
525 sort_by_field(pred.needed, [&pred](size_t i) -> size_t { return pred.next_cache_data[i].data_offset; });
526
527#ifndef TATAMI_HDF5_PARALLEL_LOCK
528 #pragma omp critical
529 {
530#else
531 TATAMI_HDF5_PARALLEL_LOCK([&]() -> void {
532#endif
533
534 size_t sofar = 0;
535 hsize_t combined_len = 0;
536 work.dataspace.selectNone();
537
538 while (sofar < pred.needed.size()) {
539 auto& first = pred.next_cache_data[pred.needed[sofar]];
540 first.mem_offset = dest_offset + combined_len;
541 hsize_t src_offset = first.data_offset;
542 hsize_t len = first.length;
543 ++sofar;
544
545 // Finding the stretch of consecutive extractions, and bundling them into a single hyperslab.
546 for (; sofar < pred.needed.size(); ++sofar) {
547 auto& next = pred.next_cache_data[pred.needed[sofar]];
548 if (src_offset + len < next.data_offset) {
549 break;
550 }
551 next.mem_offset = first.mem_offset + len;
552 len += next.length;
553 }
554
555 work.dataspace.selectHyperslab(H5S_SELECT_OR, &len, &src_offset);
556 combined_len += len;
557 }
558
559 work.memspace.setExtentSimple(1, &combined_len);
560 work.memspace.selectAll();
561 work.index.read(pred.cache_index.data() + dest_offset, HDF5::define_mem_type<Index_>(), work.memspace, work.dataspace);
562 if (needs_value) {
563 work.data.read(pred.cache_value.data() + dest_offset, HDF5::define_mem_type<Value_>(), work.memspace, work.dataspace);
564 }
565
566#ifndef TATAMI_HDF5_PARALLEL_LOCK
567 }
568#else
569 });
570#endif
571 }
572
573 pred.cache_data.swap(pred.next_cache_data);
574 pred.cache_exists.swap(pred.next_cache_exists);
575 pred.predictions_fulfilled = 1; // using the first one now.
576 return Extracted(pred, pred.predictions_made.front(), needs_value);
577 }

583 template<class Function_>
584 void extract_primary_raw(size_t i, Function_ fill, Index_ start, PrimaryWorkspace& work, bool needs_value) const {
585 Extracted details;
586 if (work.futurist) {
587 details = extract_primary_with_oracle(work, needs_value);
588 } else {
589 details = extract_primary_without_oracle(i, work, needs_value);
590 }
591
592 auto istart = details.index;
593 auto iend = details.index + details.length;
594 size_t offset = 0;
595
596 // If we used the extraction_bounds during extraction, there's no need
597 // to do another search. Similarly, if we didn't use the extraction_bounds
598 // (e.g., it was already cached) but we have extraction_bounds available,
599 // we can again skip the binary search.
600 if (!details.bounded && start) {
601 bool hit = false;
602 if (work.extraction_bounds.size()) {
603 auto& target = work.extraction_bounds[i];
604 if (target.first != -1) {
605 hit = true;
606 offset = target.first - pointers[i];
607 istart += offset;
608 }
609 }
610
611 if (!hit) {
612 istart = std::lower_bound(details.index, iend, start);
613 offset = istart - details.index;
614 }
615 }
616
617 size_t iterated = fill(istart, iend, (needs_value ? details.value + offset : NULL));
618
619 if (work.extraction_bounds.size()) {
620 auto& target = work.extraction_bounds[i];
621 if (target.first == -1) {
622 target.first = pointers[i] + offset;
623 target.second = iterated;
624 }
625 }
626
627 return;
628 }

697 if (needs_value) {

712 }

724 ++istart;

800 ************ Secondary extraction ************

802private:
803 // This could be improved by extracting multiple rows at any given call and
804 // caching them for subsequent requests. However, even then, we'd require
805 // multiple re-reads from file when we exceed the cache. So, any caching
806 // would be just turning an extremely bad access pattern into a very bad
807 // pattern, when users shouldn't even be calling this at all...
808 struct SecondaryWorkspace {
809 H5::H5File file;
810 H5::DataSet data, index;
811 H5::DataSpace dataspace;
812 H5::DataSpace memspace;

814 void fill(const HDF5CompressedSparseMatrix* parent) {
815 // TODO: set more suitable chunk cache values here, to avoid re-reading
816 // chunks on the boundaries of the primary cache.
817 file.openFile(parent->file_name, H5F_ACC_RDONLY);
818
819 data = file.openDataSet(parent->data_name);
820 index = file.openDataSet(parent->index_name);
821 dataspace = data.getSpace();
822 }
823
824 std::vector<Index_> index_cache;

827 template<class Function_>
828 bool extract_secondary_raw(Index_ primary, Index_ secondary, Function_& fill, SecondaryWorkspace& core, bool needs_value) const {
829 hsize_t left = pointers[primary], right = pointers[primary + 1];
830 core.index_cache.resize(right - left);
831
832 // Serial locks should be applied by the callers.
833 hsize_t offset = left;
834 hsize_t count = core.index_cache.size();
835 core.dataspace.selectHyperslab(H5S_SELECT_SET, &count, &offset);
836 core.memspace.setExtentSimple(1, &count);
837 core.memspace.selectAll();
838 core.index.read(core.index_cache.data(), HDF5::define_mem_type<Index_>(), core.memspace, core.dataspace);
839
840 auto it = std::lower_bound(core.index_cache.begin(), core.index_cache.end(), secondary);
841 if (it != core.index_cache.end() && *it == secondary) {
842 if (needs_value) {
843 offset = left + (it - core.index_cache.begin());
844 count = 1;
845 core.dataspace.selectHyperslab(H5S_SELECT_SET, &count, &offset);
846 core.memspace.setExtentSimple(1, &count);
847 core.memspace.selectAll();
848
849 Value_ dest;
850 core.data.read(&dest, HDF5::define_mem_type<Value_>(), core.memspace, core.dataspace);
851 fill(primary, dest);
852 } else {
853 fill(primary, 0);
854 }
855 return true;
856 } else {
857 return false;
858 }
