21void delayed_compare_run_simple(const InputValue_* input, Index_ length, Scalar_ scalar, OutputValue_* output) {
22 if constexpr(std::is_same<InputValue_, OutputValue_>::value) {
23 input = output; // basically an assertion to the compiler to enable optimizations.
24 }
25 for (Index_ i = 0; i < length; ++i) {
26 output[i] = delayed_compare<op_>(input[i], scalar);

232 my_vector(std::move(vector)),
233 my_by_row(by_row)
234 {
235 for (auto x : my_vector) {
236 if (!delayed_compare_actual_sparse<op_, InputValue_>(x)) {
237 my_sparse = false;
238 break;
239 }
240 }
241 }

255 }

262 return !my_by_row;
263 }

266 void dense(bool row, Index_ idx, Index_ start, Index_ length, const InputValue_* input, OutputValue_* output) const {
267 if (row == my_by_row) {
268 delayed_compare_run_simple<op_, InputValue_>(input, length, my_vector[idx], output);
269 } else {
270 if constexpr(std::is_same<InputValue_, OutputValue_>::value) {
271 input = output; // basically an assertion to the compiler to enable optimizations.
272 }
273 for (Index_ i = 0; i < length; ++i) {

275 }
276 }

280 if (row == my_by_row) {

289 }

298 void sparse(bool row, Index_ idx, Index_ number, const InputValue_* input_value, const Index_* indices, OutputValue_* output_value) const {
299 if (row == my_by_row) {
300 delayed_compare_run_simple<op_, InputValue_>(input_value, number, my_vector[idx], output_value);
301 } else {
302 if constexpr(std::is_same<InputValue_, OutputValue_>::value) {
303 input_value = output_value; // basically an assertion to the compiler to enable optimizations.
304 }
305 for (Index_ i = 0; i < number; ++i) {
306 output_value[i] = delayed_compare<op_, InputValue_>(input_value[i], my_vector[indices[i]]);
307 }
308 }
309 }
310
311 OutputValue_ fill(bool row, Index_ idx) const {
312 if (row == my_by_row) {
313 return delayed_compare<op_, InputValue_>(0, my_vector[idx]);
314 } else {
315 // We should only get to this point if it's sparse, otherwise no
316 // single fill value would work across the length of my_vector.
317 return 0;
318 }
