9using Stored = typename std::remove_reference<decltype(std::declval<Storage_>()[0])>::type;

11template<typename Index_, typename StoredIndex_, typename CustomPointer_, class CustomPointerModifier_>
12struct CompressedSparseSecondaryExtractorBasic {
13private:
14 StoredIndex_ max_index;
15
16 // The current position of the pointer at each primary element.
17 std::vector<CustomPointer_> current_indptrs;
18
19 // Whether to move forward or back.
20 bool lower_bound = true;
21
22 // If 'lower_bound = true', this vector contains the current index being pointed to, i.e., 'current_indices[i] := indices[current_indptrs[i]]'.
23 // We store this here as it is more cache-friendly than doing a look-up to 'indices'.
24 // If 'current_indptrs[i]' is out of range, the current index is instead set to the maximum index (e.g., max rows for CSC matrices).
25 //
26 // If 'lower_bound = false', this vector instead contains:
27 // - 'indices[current_indptrs[i] - 1]', if 'current_indptrs[i]' does not lie at the start of the primary dimension element.
28 // - otherwise, an undefined value.
29 std::vector<StoredIndex_> current_indices;
30
31 // Closest value in 'current_indices' to the 'last_request', to see whether we can short-circuit the iteration.
32 // If 'lower_bound = false', this is set to 'max_index' if all values in 'current_indices' are undefined.
33 StoredIndex_ closest_current_index;
34
35 // What was the last requested index on the secondary dimension?
36 StoredIndex_ last_request = 0;
37
38public:
39 CompressedSparseSecondaryExtractorBasic() = default;

41 template<class IndexStorage_, class PointerStorage_>
42 CompressedSparseSecondaryExtractorBasic(StoredIndex_ mi, const IndexStorage_& idx, const PointerStorage_& idp, Index_ start, Index_ length) :
43 max_index(mi), current_indices(length), current_indptrs(idp.begin() + start, idp.begin() + start + length)
44 {
45 /* Here, the general idea is to store a local copy of the actual
46 * row indices (for CSC matrices; column indices, for CSR matrices)
47 * so that we don't have to keep on doing cache-unfriendly look-ups
48 * for the indices based on the pointers that we do have. This assumes
49 * that the density is so low that updates to the local indices are
50 * rare relative to the number of comparisons to those same indices.
51 */
52 auto idpIt = idp.begin() + start;
53 for (Index_ i = 0; i < length; ++i, ++idpIt) {
54 current_indices[i] = (*idpIt < *(idpIt + 1) ? idx[*idpIt] : max_index);
55 }
56 closest_current_index = (length ? *std::min_element(current_indices.begin(), current_indices.end()) : max_index);
57 return;
58 }

64 template<class IndexStorage_, class PointerStorage_>
65 CompressedSparseSecondaryExtractorBasic(StoredIndex_ mi, const IndexStorage_& idx, const PointerStorage_& idp, const Index_* subset, Index_ length) :
66 max_index(mi), current_indices(length), current_indptrs(length)
67 {
68 for (Index_ i0 = 0; i0 < length; ++i0) {
69 auto i = subset[i0];
70 current_indptrs[i0] = idp[i];
71 current_indices[i0] = (idp[i] < idp[i + 1] ? idx[idp[i]] : max_index);
72 }
73 closest_current_index = (length ? *std::min_element(current_indices.begin(), current_indices.end()) : max_index);
74 return;
75 }

78 template<class IndexStorage_, class PointerStorage_, class StoreFunction_, class SkipFunction_>
79 void search_above_or_equal(
80 StoredIndex_ secondary,
81 Index_ index_primary,
82 Index_ primary,
83 const IndexStorage_& indices,
84 const PointerStorage_& indptrs,
85 StoreFunction_& store,
86 SkipFunction_& skip
87 ) {
88 auto& curdex = current_indices[index_primary];
89
90 // Fixing this if we were previously iterating in the wrong direction.
91 if (!lower_bound) {
92 auto limit = indptrs[primary + 1];
93 auto curptr = current_indptrs[index_primary];
94 auto raw_ptr = CustomPointerModifier_::get(curptr);
95 if (raw_ptr != limit) {
96 curdex = indices[raw_ptr];
97 } else {
98 curdex = max_index;
99 }
100 }
101
102 // Skipping if the curdex (corresponding to curptr) is already higher
103 // than secondary. So, we only need to do more work if the request is
104 // greater than the stored index. This also catches cases where we're
105 // at the end of the dimension, as curdex is set to max_index.
106 if (curdex > secondary) {
107 closest_current_index = std::min(curdex, closest_current_index);
108 skip(primary);
109 return;
110 }
111
112 auto& curptr = current_indptrs[index_primary];
113 if (curdex == secondary) {
114 closest_current_index = secondary;
115 store(primary, curptr);
116 return;
117 }
118
119 auto limit = indptrs[primary + 1];
120
121 // Special case if the requested index is at the end of the matrix, in
122 // which case we can just jump there directly rather than doing an
123 // unnecessary binary search. Note that 'limit - 1' is always available
124 // as this dimension element should be non-empty if secondary > curdex.
125 if (secondary + 1 == max_index) {
126 if (indices[limit - 1] == secondary) {
127 CustomPointerModifier_::set(curptr, limit); // don't set directly to 'limit - 1' as this won't be the start of the run in semi-compressed mode.
128 CustomPointerModifier_::decrement(curptr, indices, indptrs[primary]);
129 curdex = secondary;
130 closest_current_index = secondary;
131 store(primary, curptr);
132 } else {
133 CustomPointerModifier_::set(curptr, limit);
134 curdex = max_index;
135 skip(primary);
136 }
137 return;
138 }
139
140 // Having a peek at the index of the next non-zero element; maybe we're
141 // lucky enough that the requested index is below this, as would be the
142 // case for consecutive or near-consecutive accesses.
143 CustomPointerModifier_::increment(curptr, indices, limit);
144 auto raw_ptr = CustomPointerModifier_::get(curptr);
145 if (raw_ptr == limit) {
146 curdex = max_index;
147 skip(primary);
148 return;
149 }
150
151 curdex = indices[raw_ptr];
152 if (curdex > secondary) {
153 closest_current_index = std::min(curdex, closest_current_index);
154 skip(primary);
155 return;
156 }
157
158 if (curdex == secondary) {
159 closest_current_index = secondary;
160 store(primary, curptr);
161 return;
162 }
163
164 // Otherwise we need to search indices above the existing position.
165 // We do a quick increment to cut down the search space; don't
166 // need to pay the cost of using increment() here, as the lower
167 // bound search is going to be faster than any increment.
168 ++raw_ptr;
169 Stored<PointerStorage_> next_ptr = std::lower_bound(indices.begin() + raw_ptr, indices.begin() + limit, secondary) - indices.begin();
170 CustomPointerModifier_::set(curptr, next_ptr);
171
172 if (next_ptr == limit) {
173 curdex = max_index;
174 skip(primary);
175 return;
176 }
177
178 curdex = indices[next_ptr];
179 if (curdex > secondary) {
180 closest_current_index = std::min(curdex, closest_current_index);
181 skip(primary);
182 return;
183 }
184
185 closest_current_index = secondary;
186 store(primary, curptr);
187 return;
188 }

197 template<class IndexStorage_, class PointerStorage_, class StoreFunction_, class SkipFunction_>
198 void search_below(
199 StoredIndex_ secondary,
200 Index_ index_primary,
201 Index_ primary,
202 const IndexStorage_& indices,
203 const PointerStorage_& indptrs,
204 StoreFunction_& store,
205 SkipFunction_& skip
206 ) {
207 auto& curptr = current_indptrs[index_primary];
208
209 // Can't decrement anymore, in which case we quit. Note that there's
210 // no need to check if 'indices[curptr] == secondary'; we only enter
211 // this function if 'last_request > secondary', and we know that
212 // 'indices[curptr] >= last_request > secondary'.
213 auto lower_limit = indptrs[primary];
214 auto raw_ptr = CustomPointerModifier_::get(curptr);
215 if (raw_ptr == lower_limit) {
216 skip(primary);
217 return;
218 }
219
220 // Checking if the next-lowest index is below the requested
221 // 'secondary', at which point we can just quit without a
222 // cache-unfriendly lookup to 'indices'. This is valid as we know that
223 // we are not at the start of the dimension at this point.
224 auto& curdex = current_indices[index_primary];
225 if (!lower_bound) {
226 if (curdex < secondary) {
227 update_closest_index(curdex);
228 skip(primary);
229 return;
230 }
231 }
232
233 // Special case if the requested index is at the end of the matrix, in
234 // which case we can just jump there directly rather than doing an
235 // unnecessary binary search.
236 if (secondary == 0) {
237 auto first_ptr = lower_limit;
238 CustomPointerModifier_::set(curptr, first_ptr);
239
240 if (indices[first_ptr] == secondary) {
241 closest_current_index = secondary;
242 store(primary, curptr);
243 } else {
244 skip(primary);
245 }
246 return;
247 }
248
249 // Having a peek at the index of the next non-zero element and seeing
250 // whether we can stop searching, as would be the case for consecutive
251 // or near-consecutive accesses.
252 --raw_ptr;
253 auto candidate = indices[raw_ptr];
254 if (candidate < secondary) {
255 curdex = candidate;
256 update_closest_index(curdex);
257 skip(primary);
258 return;
259 }
260
261 if (candidate == secondary) {
262 CustomPointerModifier_::decrement(curptr, indices, lower_limit);
263 closest_current_index = secondary;
264 store(primary, curptr);
265 return;
266 }
267
268 // Otherwise, searching indices below the (just-decremented) position.
269 Stored<PointerStorage_> next_ptr = std::lower_bound(indices.begin() + lower_limit, indices.begin() + raw_ptr, secondary) - indices.begin();
270 CustomPointerModifier_::set(curptr, next_ptr);
271
272 if (next_ptr == indptrs[primary + 1]) {
273 skip(primary);
274 return;
275 }
276
277 if (indices[next_ptr] == secondary) {
278 closest_current_index = secondary;
279 store(primary, curptr);
280 return;
281 }
282
283 if (next_ptr == lower_limit) {
284 skip(primary);
285 return;
286 }
287
288 curdex = indices[next_ptr - 1]; // cheap decrement to inspect the next-lowest element.
289 update_closest_index(curdex);
290 skip(primary);
291 return;
292 }

295 template<class IndexStorage_, class PointerStorage_, class PrimaryFunction_, class StoreFunction_, class SkipFunction_>
296 bool search(
297 StoredIndex_ secondary,
298 Index_ primary_length,
299 PrimaryFunction_ to_primary,
300 const IndexStorage_& indices,
301 const PointerStorage_& indptrs,
302 StoreFunction_ store,
303 SkipFunction_ skip
304 ) {
305 if (secondary >= last_request) {
306 if (lower_bound && secondary < closest_current_index) {
307 return false;
308 }
309
310 closest_current_index = max_index; // resetting this.
311 for (Index_ p = 0; p < primary_length; ++p) {
312 search_above_or_equal(secondary, p, to_primary(p), indices, indptrs, store, skip);
313 }
314
315 lower_bound = true;
316
317 } else {
318 if (!lower_bound && (closest_current_index == max_index || secondary > closest_current_index)) {
319 return false;
320 }
321
322 closest_current_index = max_index; // resetting this.
323 for (Index_ p = 0; p < primary_length; ++p) {
324 search_below(secondary, p, to_primary(p), indices, indptrs, store, skip);
325 }
326
327 lower_bound = false;
328 }
329
330 last_request = secondary;
331 return true;
332 }
