113 const std::vector<Index_>& cumulative,
114 bool needs_value,
115 bool needs_index,
116 Value_* vbuffer,
117 Index_* ibuffer,
118 Store_ store)
119{
120 auto vcopy = vbuffer;
121 auto icopy = ibuffer;
122 Index_ accumulated = 0;
123
124 for (size_t x = 0, end = cumulative.size() - 1; x < end; ++x) {
125 auto range = store(x, vcopy, icopy);
126 accumulated += range.number;
127 if (needs_value) {
128 copy_n(range.value, range.number, vcopy);
129 vcopy += range.number;
130 }
131 if (needs_index) {
132 auto offset = cumulative[x];

136 icopy += range.number;
137 }
138 }
139
140 return SparseRange<Value_, Index_>(
141 accumulated,
142 (needs_value ? vbuffer : NULL),

231template<typename Value_, typename Index_, class Initialize_>
232size_t initialize_parallel_block(
233 const std::vector<Index_>& cumulative,
234 const std::vector<std::shared_ptr<const Matrix<Value_, Index_> > >& mats,
235 Index_ block_start,
236 Index_ block_length,
237 Initialize_ init)
238{
239 size_t start_index = find_segment(cumulative, block_start); // finding the first matrix.
240 Index_ actual_start = block_start - cumulative[start_index];
241 Index_ block_end = block_start + block_length;
242
243 size_t nmats = mats.size();
244 for (auto index = start_index; index < nmats; ++index) {
245 bool not_final = (block_end > cumulative[index + 1]);
246 Index_ actual_end = (not_final ? cumulative[index + 1] : block_end) - cumulative[index];
247 init(index, actual_start, actual_end - actual_start);
248 if (!not_final) {
249 break;
250 }
251 actual_start = 0;
252 }
253
254 return start_index;
255}

494template<typename Value_, typename Index_, class Initialize_>
495void initialize_parallel_index(
496 const std::vector<Index_>& cumulative,
497 const std::vector<std::shared_ptr<const Matrix<Value_, Index_> > >& mats,
498 const std::vector<Index_>& indices,
499 Initialize_ init)
500{
501 if (indices.empty()) {
502 return;
503 }
504 size_t start_index = find_segment(cumulative, indices.front());
505 size_t nmats = cumulative.size();
506
507 Index_ counter = 0, il = indices.size();
508 for (size_t index = 0; index < nmats; ++index) {
509 Index_ lower = cumulative[index];
510 Index_ upper = cumulative[index + 1];
511
512 std::vector<Index_> curslice;
513 while (counter < il && indices[counter] < upper) {
514 curslice.push_back(indices[counter] - lower);
515 ++counter;
516 }
517
518 if (!curslice.empty()) {
519 init(index, std::move(curslice));
520 }
521
522 if (counter == il) {
523 break;
524 }
525 }
526}

756struct ChooseSegment {
757 ChooseSegment(const std::vector<Index_>& cum) : cumulative(cum) {}
758
759 const std::vector<Index_>& cumulative;
760private:
761 size_t last_segment = 0;
762
763public:
764 size_t choose_segment(Index_ i) {
765 if (cumulative[last_segment] > i) {
766 if (last_segment && cumulative[last_segment - 1] <= i) {
767 --last_segment;
768 } else {
769 last_segment = find_segment(cumulative, i);
770 }
771 } else if (cumulative[last_segment + 1] <= i) {
772 if (last_segment + 2 < cumulative.size() && cumulative[last_segment + 2] > i) {
773 ++last_segment;
774 } else {
775 last_segment = find_segment(cumulative, i);
776 }
777 }
778 return last_segment;
779 }

834template<typename Index_, class Initialize_>
835void initialize_perp_oracular(
836 const std::vector<Index_>& cumulative,
837 const Oracle<Index_>* oracle,
838 std::vector<size_t>& chosen,
839 Initialize_ init)
840{
841 size_t ntotal = oracle->total();
842 chosen.reserve(ntotal);
843 ChooseSegment<Index_> chooser(cumulative);
844
845 struct Predictions {
846 bool consecutive = true;
847 Index_ start = 0;
848 Index_ number = 0;
849 std::vector<Index_> predictions;
850
851 void add(Index_ p) {
852 if (consecutive) {
853 if (number == 0) {
854 start = p;
855 number = 1;
856 return;
857 }
858 if (number + start == p) {
859 ++number;
860 return;
861 }
862 consecutive = false;
863 predictions.resize(number);
864 std::iota(predictions.begin(), predictions.end(), start);
865 }
866
867 predictions.push_back(p);
868 }
869 };
870
871 size_t nmats = cumulative.size() - 1;
872 std::vector<Predictions> predictions(nmats);
873 for (size_t i = 0; i < ntotal; ++i) {
874 auto prediction = oracle->get(i);
875 size_t choice = chooser.choose_segment(prediction);
876 chosen.push_back(choice);
877 predictions[choice].add(prediction - cumulative[choice]);
878 }
879
880 for (size_t x = 0; x < nmats; ++x) {
881 auto& current = predictions[x];
882 if (current.consecutive) {
883 if (current.number) {
884 init(x, std::make_shared<ConsecutiveOracle<Index_> >(current.start, current.number));
885 }
886 } else {
887 if (!current.predictions.empty()) {
888 init(x, std::make_shared<FixedVectorOracle<Index_> >(std::move(current.predictions)));
889 }
890 }
891 }
892}

982template<int margin_, typename Value_, typename Index_>
983class DelayedBind : public Matrix<Value_, Index_> {
984public:
989 DelayedBind(std::vector<std::shared_ptr<const Matrix<Value_, Index_> > > ps) : mats(std::move(ps)), cumulative(mats.size()+1) {
990 size_t sofar = 0;
991 for (size_t i = 0, nmats = mats.size(); i < nmats; ++i) {
992 auto& current = mats[i];
993 Index_ primary, secondary;
994 if constexpr(margin_ == 0) {
995 primary = current->nrow();
996 secondary = current->ncol();
997 } else {
998 primary = current->ncol();
999 secondary = current->nrow();
1000 }
1001
1002 if (i == 0) {
1003 otherdim = secondary;
1004 } else if (otherdim != secondary) {
1005 throw std::runtime_error("all 'mats' should have the same number of " + (margin_ == 0 ? std::string("columns") : std::string("rows")));
1006 }
1007
1008 // Removing the matrices that don't contribute anything,
1009 // so we don't have to deal with their overhead.
1010 if (primary > 0) {
1011 if (sofar != i) {
1012 mats[sofar] = std::move(current);
1013 }
1014 cumulative[sofar + 1] = cumulative[sofar] + primary;
1015 ++sofar;
1016 }
1017 }
1018
1019 cumulative.resize(sofar + 1);
1020 mats.resize(sofar);
1021
1022 double denom = 0;
1023 for (const auto& x : mats) {
1024 double total = x->nrow() * x->ncol();
1025 denom += total;
1026 sparse_prop += total * x->sparse_proportion();
1027 row_prop += total * x->prefer_rows_proportion();
1028 }
1029 if (denom) {
1030 sparse_prop /= denom;
1031 row_prop /= denom;
1032 }
1033
1034 for (int d = 0; d < 2; ++d) {
1035 stored_uses_oracle[d] = false;
1036 for (const auto& x : mats) {
1037 if (x->uses_oracle(d)) {
1038 stored_uses_oracle[d] = true;
1039 break;
1040 }
1041 }
1042 }
1043 }

1049 DelayedBind(const std::vector<std::shared_ptr<Matrix<Value_, Index_> > >& ps) : DelayedBind(std::vector<std::shared_ptr<const Matrix<Value_, Index_> > >(ps.begin(), ps.end())) {}

1051private:
1052 std::vector<std::shared_ptr<const Matrix<Value_, Index_> > > mats;
1053 Index_ otherdim = 0;
1054 std::vector<Index_> cumulative;
1055
1056 double sparse_prop = 0, row_prop = 0;
1057 std::array<bool, 2> stored_uses_oracle;
1058
1059public:

1061 if constexpr(margin_==0) {
1062 return cumulative.back();
1063 } else {
1064 return otherdim;
1065 }
