261 template<bool ROW, typename ExtractType>
262 const T* extract(size_t primary_start, size_t primary_length, T* target, const ExtractType& extract_value, size_t extract_length, Hdf5WorkspaceBase& work) const {
263 hsize_t offset[2];
264 hsize_t count[2];
265
266 constexpr int dimdex = (ROW != transpose);
267 offset[1-dimdex] = primary_start;
268 count[1-dimdex] = primary_length;
269
270 constexpr bool indexed = std::is_same<ExtractType, std::vector<IDX> >::value;
271
272 if constexpr(indexed) {
273 // Take slices across the current chunk for each index. This should be okay if consecutive,
274 // but hopefully they've fixed the problem with non-consecutive slices in:
275 // https://forum.hdfgroup.org/t/union-of-non-consecutive-hyperslabs-is-very-slow/5062
276 count[dimdex] = 1;
277 work.dataspace.selectNone();
278 for (auto idx : extract_value) {
279 offset[dimdex] = idx;
280 work.dataspace.selectHyperslab(H5S_SELECT_OR, count, offset);
281 }
282 count[dimdex] = extract_length; // for the memspace setter.
283 } else {
284 offset[dimdex] = extract_value;
285 count[dimdex] = extract_length;
286 work.dataspace.selectHyperslab(H5S_SELECT_SET, count, offset);
287 }
288
289 // HDF5 is a lot faster when the memspace and dataspace match in dimensionality.
290 // Presumably there is some shuffling that happens inside when dimensions don't match.
291 work.memspace.setExtentSimple(2, count);
292 work.memspace.selectAll();
293
294 work.dataset.read(target, HDF5::define_mem_type<T>(), work.memspace, work.dataspace);
295 }

297 template<bool ROW, typename ExtractType>
298 const T* extract(size_t i, T* buffer, const ExtractType& extract_value, size_t extract_length, Hdf5WorkspaceBase& work) const {
299 // Figuring out which chunk the request belongs to.
300 hsize_t cache_mydim, mydim, otherdim;
301 if constexpr(ROW != transpose) {
302 cache_mydim = cache_firstdim;
303 mydim = firstdim;
304 otherdim = seconddim;
305 } else {
306 cache_mydim = cache_seconddim;
307 mydim = seconddim;
308 otherdim = firstdim;
309 }
310
311 // No caching can be done here, so we just extract directly.
312 if (cache_mydim == 0) {
313 extract<ROW>(i, 1, buffer, extract_value, extract_length, work);
314 return buffer;
315 }
316
317 size_t chunk = i / cache_mydim;
318 if (chunk != work.cached_chunk || work.init) {
319 work.init = false;
320
321 hsize_t cache_mydim_start = chunk * cache_mydim;
322 hsize_t cache_mydim_end = std::min(mydim, cache_mydim_start + cache_mydim);
323 hsize_t cache_mydim_actual = cache_mydim_end - cache_mydim_start;
324 size_t dense_cache_size = extract_length * cache_mydim_actual;
325
326 T* destination;
327 work.cache.resize(dense_cache_size);
328 if constexpr(ROW != transpose) {
329 destination = work.cache.data();
330 } else {
331 work.buffer.resize(dense_cache_size);
332 destination = work.buffer.data();
333 }
334
335 extract<ROW>(cache_mydim_start, cache_mydim_actual, destination, extract_value, extract_length, work);
336
337 if constexpr(ROW == transpose) {
338 auto output = work.cache.begin();
339 for (hsize_t x = 0; x < cache_mydim_actual; ++x, output += extract_length) {
340 auto in = work.buffer.begin() + x;
341 for (hsize_t y = 0; y < extract_length; ++y, in += cache_mydim_actual) {
342 *(output + y) = *in;
343 }
344 }
345 }
346
347 work.cached_chunk = chunk;
348 }
349
350 size_t index = i % cache_mydim;
351 auto wIt = work.cache.begin() + index * extract_length;
352 std::copy(wIt, wIt + extract_length, buffer);
353 return buffer;
354 }
