187 // Turn off HDF5's caching, as we'll be handling that. This allows us
188 // to parallelize extractions without locking when the data has already
189 // been loaded into memory; if we just used HDF5's cache, we would have
190 // to lock on every extraction, given the lack of thread safety.

196 dataspace = dataset.getSpace();

199 per_cache_size = static_cast<size_t>(chunk_dim) * static_cast<size_t>(other_dim);
200 num_chunks = static_cast<double>(parent->total_cache_size) / per_cache_size;
201
202 historian.reset(new LruCache);

226 template<bool accrow_, typename ExtractType_>
227 static void extract_base(Index_ primary_start, Index_ primary_length, Value_* target, const ExtractType_& extract_value, Index_ extract_length, Workspace<accrow_>& work) {
228 hsize_t offset[2];
229 hsize_t count[2];
230
231 constexpr int dimdex = (accrow_ != transpose_);
232 offset[1-dimdex] = primary_start;
233 count[1-dimdex] = primary_length;
234
235 constexpr bool indexed = std::is_same<ExtractType_, std::vector<Index_> >::value;
236
237 if constexpr(indexed) {
238 // Take slices across the current chunk for each index. This should be okay if consecutive,
239 // but hopefully they've fixed the problem with non-consecutive slices in:
240 // https://forum.hdfgroup.org/t/union-of-non-consecutive-hyperslabs-is-very-slow/5062
241 count[dimdex] = 1;
242 work.dataspace.selectNone();
243 for (auto idx : extract_value) {
244 offset[dimdex] = idx;
245 work.dataspace.selectHyperslab(H5S_SELECT_OR, count, offset);
246 }
247 count[dimdex] = extract_length; // for the memspace setter.
248 } else {
249 offset[dimdex] = extract_value;
250 count[dimdex] = extract_length;
251 work.dataspace.selectHyperslab(H5S_SELECT_SET, count, offset);
252 }
253
254 // HDF5 is a lot faster when the memspace and dataspace match in dimensionality.
255 // Presumably there is some shuffling that happens inside when dimensions don't match.
256 work.memspace.setExtentSimple(2, count);
257 work.memspace.selectAll();
258
259 work.dataset.read(target, HDF5::define_mem_type<Value_>(), work.memspace, work.dataspace);
260 }

262 template<bool accrow_, typename ExtractType_>
263 static Index_ extract_chunk(Index_ chunk_id, Index_ dim, Index_ chunk_dim, Value_* target, const ExtractType_& extract_value, Index_ extract_length, Workspace<accrow_>& work) {
264 Index_ chunk_start = chunk_id * chunk_dim;
265 Index_ chunk_end = std::min(dim, chunk_start + chunk_dim);
266 Index_ chunk_actual = chunk_end - chunk_start;
267 extract_base<accrow_>(chunk_start, chunk_actual, target, extract_value, extract_length, work);
268 return chunk_actual;
269 }

271 static void transpose(std::vector<Value_>& cache, std::vector<Value_>& buffer, Index_ actual_dim, Index_ extract_length) {
272 buffer.resize(cache.size());
273 auto output = buffer.begin();
274 for (Index_ x = 0; x < actual_dim; ++x, output += extract_length) {
275 auto in = cache.begin() + x;
276 for (Index_ y = 0; y < extract_length; ++y, in += actual_dim) {
277 *(output + y) = *in;
278 }
279 }
280 cache.swap(buffer);
281 return;
282 }

305 template<bool accrow_, typename ExtractType_>
306 const Value_* extract_with_oracle(Index_ mydim, Index_ chunk_mydim, const ExtractType_& extract_value, Index_ extract_length, Workspace<accrow_>& work) const {
307 auto& pred = *work.futurist;
308 if (pred.predictions_made.size() > pred.predictions_fulfilled) {
309 const auto& chosen = pred.predictions_made[pred.predictions_fulfilled++];
310 return pred.cache_data[chosen.first].data() + extract_length * chosen.second;
311 }
312
313 /* We use the oracle to batch together multiple HDF5 library
314 * calls so that we don't have to reenter the critical section
315 * for every fetch() call. Extracted data are cached for
316 * easy parallel retrieval without involving the HDF5 library.
317 *
318 * We don't bother trying to bundle everything into a single
319 * HDF5 call, as this is a pain to pull data out into the
320 * individual cache buffers afterwards.
321 */
322 Index_ num_chunks = work.num_chunks;
323 Index_ used = 0;
324
325 bool starting = pred.cache_data.empty();
326 if (starting) {
327 pred.cache_data.resize(num_chunks);
328 pred.next_cache_data.resize(num_chunks, std::vector<Value_>(1)); // using a placeholder value for availability checks.
329 pred.chunks_in_need.reserve(num_chunks);
330 } else {
331 pred.next_cache_exists.clear();
332 pred.chunks_in_need.clear();
333 }
334
335 size_t max_predictions = static_cast<size_t>(num_chunks) * chunk_mydim * 2; // double the cache size, basically.
336 pred.predictions_made.clear();
337 pred.predictions_made.reserve(max_predictions);
338
339 for (size_t p = 0; p < max_predictions; ++p) {
340 Index_ current;
341 if (!pred.prediction_stream.next(current)) {
342 break;
343 }
344
345 auto curchunk = current / chunk_mydim;
346 auto curindex = current % chunk_mydim;
347
348 auto it = pred.next_cache_exists.find(curchunk);
349 if (it == pred.next_cache_exists.end()) {
350 pred.next_cache_exists[curchunk] = used;
351 pred.predictions_made.emplace_back(used, curindex);
352
353 auto it2 = pred.cache_exists.find(curchunk);
354 if (it2 != pred.cache_exists.end()) {
355 pred.next_cache_data[used].swap(pred.cache_data[it2->second]);
356 } else {
357 pred.chunks_in_need.emplace_back(curchunk, used);
358 }
359
360 ++used;
361 if (used == num_chunks) {
362 break;
363 }
364 } else {
365 pred.predictions_made.emplace_back(it->second, curindex);
366 }
367 }
368
381 size_t search = 0;
382 for (const auto& c : pred.chunks_in_need) {
383 // If we're lucky enough to already be sitting on an allocation
384 // (e.g., if the last call to this function didn't consume all
385 // cache elements), we just use it directly.
386 if (pred.next_cache_data[c.second].empty()) {
387 while (pred.cache_data[search].empty()) {
388 ++search;
389 }
390
391#ifdef DEBUG
392 // This should never reach the end, as there should always be
393 // 'num_chunks' non-empty elements across cache_data and
394 // next_cache_data. Nonetheless, we add an assertion here.
395 if (search >= pred.cache_data.size()) {
396 throw std::runtime_error("internal cache management error for HDF5DenseMatrix with oracles");
397 }
398#endif
399
400 pred.next_cache_data[c.second].swap(pred.cache_data[search]);
401 ++search;
402 }
403
404 pred.next_cache_data[c.second].resize(work.per_cache_size); // eventually no-op when all available caches are of the right size.
405 }
406
407 if constexpr(accrow_ == transpose_) {
408 pred.cache_transpose_info.clear();
409 }
410
411#ifndef TATAMI_HDF5_PARALLEL_LOCK
412 #pragma omp critical
413 {
414#else
415 TATAMI_HDF5_PARALLEL_LOCK([&]() -> void {
416#endif
417
418 for (const auto& c : pred.chunks_in_need) {
419 auto& cache_target = pred.next_cache_data[c.second];
420 auto actual_dim = extract_chunk<accrow_>(c.first, mydim, chunk_mydim, cache_target.data(), extract_value, extract_length, work);
421 if constexpr(accrow_ == transpose_) {
422 pred.cache_transpose_info.emplace_back(c.second, actual_dim);
423 }
424 }
425
426#ifndef TATAMI_HDF5_PARALLEL_LOCK
427 }
428#else
429 });
430#endif
431
432 // Applying transpositions to all cached buffers for easier retrieval, but only once the lock is released.
433 if constexpr(accrow_ == transpose_) {
434 for (const auto& x : pred.cache_transpose_info) {
435 transpose(pred.next_cache_data[x.first], work.transposition_buffer, x.second, extract_length);
436 }
437 }
438
439 pred.cache_data.swap(pred.next_cache_data);
440 pred.cache_exists.swap(pred.next_cache_exists);
441
442 pred.predictions_fulfilled = 1; // well, because we just used one.
443 const auto& chosen = pred.predictions_made.front(); // assuming at least one prediction was made, otherwise, why was fetch() even called?
444 return pred.cache_data[chosen.first].data() + extract_length * chosen.second;
445 }

447 template<bool accrow_, typename ExtractType_>
448 const Value_* extract_without_oracle(Index_ i, Index_ mydim, Index_ chunk_mydim, const ExtractType_& extract_value, Index_ extract_length, Workspace<accrow_>& work) const {
449 auto chunk = i / chunk_mydim;
450 auto index = i % chunk_mydim;
451
452 auto& historian = *(work.historian);
453 auto it = historian.cache_exists.find(chunk);
454 if (it != historian.cache_exists.end()) {
455 auto chosen = it->second;
456 historian.cache_data.splice(historian.cache_data.end(), historian.cache_data, chosen); // move to end.
457 return chosen->first.data() + index * extract_length;
458 }
459
460 // Adding a new last element, or recycling the front to the back.
461 typename std::list<typename LruCache::Element>::iterator location;
462 if (historian.cache_data.size() < static_cast<size_t>(work.num_chunks)) {
463 historian.cache_data.emplace_back(std::vector<Value_>(work.per_cache_size), chunk);
464 location = std::prev(historian.cache_data.end());
465 } else {
466 location = historian.cache_data.begin();
467 historian.cache_exists.erase(location->second);
468 location->second = chunk;
469 historian.cache_data.splice(historian.cache_data.end(), historian.cache_data, location); // move to end.
470 }
471 historian.cache_exists[chunk] = location;
472
473 auto& cache_target = location->first;
474 Index_ actual_dim;
475
476#ifndef TATAMI_HDF5_PARALLEL_LOCK
477 #pragma omp critical
478 {
479#else
480 TATAMI_HDF5_PARALLEL_LOCK([&]() -> void {
481#endif
482
483 actual_dim = extract_chunk<accrow_>(chunk, mydim, chunk_mydim, cache_target.data(), extract_value, extract_length, work);
484
485#ifndef TATAMI_HDF5_PARALLEL_LOCK
486 }
487#else
488 });
489#endif
490
491 // Applying a transposition for easier retrieval, but only once the lock is released.
492 if constexpr(accrow_ == transpose_) {
493 transpose(cache_target, work.transposition_buffer, actual_dim, extract_length);
494 }
495
496 return cache_target.data() + index * extract_length;
497 }

499 template<bool accrow_, typename ExtractType_>
500 const Value_* extract(Index_ i, Value_* buffer, const ExtractType_& extract_value, Index_ extract_length, Workspace<accrow_>& work) const {
501 // If there isn't any space for caching, we just extract directly.
502 if (work.num_chunks == 0) {
503 return extract_without_cache(i, buffer, extract_value, extract_length, work);
504 }
505
506 Index_ chunk_mydim, mydim;
507 if constexpr(accrow_ != transpose_) {
508 chunk_mydim = chunk_firstdim;
509 mydim = firstdim;
510 } else {
511 chunk_mydim = chunk_seconddim;
512 mydim = seconddim;
513 }
514
515 const Value_* cache;
516 if (work.futurist) {
517 cache = extract_with_oracle(mydim, chunk_mydim, extract_value, extract_length, work);
518 } else {
519 cache = extract_without_oracle(i, mydim, chunk_mydim, extract_value, extract_length, work);;
520 }
521
522 std::copy(cache, cache + extract_length, buffer);
523 return buffer;
524 }
