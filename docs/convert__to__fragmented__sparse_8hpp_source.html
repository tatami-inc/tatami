80template<typename StoredValue_, typename StoredIndex_, typename InputValue_, typename InputIndex_>
81FragmentedSparseContents<StoredValue_, StoredIndex_> retrieve_fragmented_sparse_contents_consistent(
82 const Matrix<InputValue_, InputIndex_>& matrix,
83 const bool row,
84 const RetrieveFragmentedSparseContentsOptions& options
85) {
86 const InputIndex_ NR = matrix.nrow();
87 const InputIndex_ NC = matrix.ncol();
88 const InputIndex_ primary = (row ? NR : NC);
89 const InputIndex_ secondary = (row ? NC : NR);
90
91 FragmentedSparseContents<StoredValue_, StoredIndex_> output(primary);
92 auto& store_v = output.value;
93 auto& store_i = output.index;
94
95 if (matrix.is_sparse()) {
96 parallelize([&](const int, const InputIndex_ start, const InputIndex_ length) -> void {
97 auto wrk = consecutive_extractor<true>(matrix, row, start, length);
98 auto buffer_v = create_container_of_Index_size<std::vector<InputValue_> >(secondary);
99 auto buffer_i = create_container_of_Index_size<std::vector<InputIndex_> >(secondary);
100
101 for (InputIndex_ p = start, pe = start + length; p < pe; ++p) {
102 const auto range = wrk->fetch(buffer_v.data(), buffer_i.data());
103 auto& sv = store_v[p];
104 auto& si = store_i[p];
105 sv.reserve(range.number);
106 si.reserve(range.number);
107
108 // We don't filter out structural non-zeros that have values of zero, for consistency with convert_to_compressed_sparse().
109 for (InputIndex_ i = 0; i < range.number; ++i) {
110 sv.push_back(range.value[i]);
111 si.push_back(range.index[i]);
112 }
113 }
114 }, primary, options.num_threads);
115
116 } else {
117 parallelize([&](const int, const InputIndex_ start, const InputIndex_ length) -> void {
118 auto wrk = consecutive_extractor<false>(matrix, row, start, length);
119 auto buffer_v = create_container_of_Index_size<std::vector<InputValue_> >(secondary);
120
121 // Special conversion from dense to save ourselves from having to make
122 // indices that we aren't really interested in.
123 for (InputIndex_ p = start, pe = start + length; p < pe; ++p) {
124 const auto ptr = wrk->fetch(buffer_v.data());
125 auto& sv = store_v[p];
126 auto& si = store_i[p];
127
128 for (InputIndex_ s = 0; s < secondary; ++s) {
129 const auto val = ptr[s];
130 if (val) {
131 sv.push_back(val);
132 si.push_back(s);
133 }
134 }
135 }
136 }, primary, options.num_threads);
137 }
138
139 return output;
140}

157CountFragmentedSparseNonZerosInconsistentResult<Index_> count_fragmented_sparse_non_zeros_inconsistent(
158 const tatami::Matrix<Value_, Index_>& matrix,
159 const Index_ primary,
160 const Index_ secondary,
161 const bool row,
162 const int threads
163) {
164 const bool do_parallel = threads > 1;
165 std::optional<std::vector<std::optional<std::vector<Index_> > > > all_partial_counts;
166 if (do_parallel) {
167 all_partial_counts.emplace(sanisizer::cast<I<decltype(all_partial_counts->size())> >(threads - 1));
168 }
169
170 const bool is_sparse = matrix.is_sparse();
171 CountFragmentedSparseNonZerosInconsistentResult<Index_> output(primary, secondary, is_sparse);
172 auto& nnz_inconsistent = output.inconsistent;
173 auto& nnz_consistent = output.consistent;
174
175 const int num_used = parallelize([&](const int thread, const Index_ start, const Index_ length) -> void {
176 // To minimize false sharing, we allocate each buffer as a per-thread vector before moving it into the nnz_workers for serial use.
177 // We skip the allocation for the first thread as this is allowed to use the (presumably zeroed) nnz array directly.
178 Index_* cur_counts;
179 std::optional<std::vector<Index_> > count_holder;
180 if (!do_parallel) {
181 cur_counts = nnz_inconsistent.data();
182 } else {
183 if (thread == 0) {
184 cur_counts = nnz_inconsistent.data();
185 } else {
186 count_holder.emplace(cast_Index_to_container_size<std::vector<Index_> >(primary));
187 cur_counts = count_holder->data();
188 }
189 }
190
191 if (is_sparse) {
192 Options opt;
193 opt.sparse_extract_value = false;
194 opt.sparse_ordered_index = false;
195 auto wrk = consecutive_extractor<true>(matrix, !row, start, length, opt);
196 auto buffer_i = create_container_of_Index_size<std::vector<Index_> >(primary);
197
198 for (Index_ x = 0; x < length; ++x) {
199 const auto range = wrk->fetch(NULL, buffer_i.data());
200 for (Index_ i = 0; i < range.number; ++i) {
201 ++cur_counts[range.index[i]];
202 }
203 }
204
205 } else {
206 auto wrk = consecutive_extractor<false>(matrix, !row, start, length);
207 auto buffer_v = create_container_of_Index_size<std::vector<Value_> >(primary);
208
209 for (Index_ x = 0; x < length; ++x) {
210 const auto ptr = wrk->fetch(buffer_v.data());
211 Index_ count = 0;
212 for (Index_ p = 0; p < primary; ++p) {
213 const bool is_nz = (ptr[p] != 0);
214 cur_counts[p] += is_nz;
215 count += is_nz;
216 }
217 (*nnz_consistent)[start + x] = count;
218 }
219 }
220
221 if (do_parallel) {
222 if (thread > 0) {
223 (*all_partial_counts)[thread - 1] = std::move(count_holder);
224 }
225 }
226 }, secondary, threads);
227
228 if (do_parallel) {
229 for (int t = 1; t < num_used; ++t) {
230 const auto& y = *((*all_partial_counts)[t - 1]);
231 for (Index_ p = 0; p < primary; ++p) {
232 nnz_inconsistent[p] += y[p];
233 }
234 }
235 }
236
237 return output;
238}

241void fill_fragmented_sparse_matrix_inconsistent(
242 const tatami::Matrix<InputValue_, InputIndex_>& matrix,
243 const InputIndex_ primary,
244 const InputIndex_ secondary,
245 const bool row,
246 FragmentedSparseContents<StoredValue_, StoredIndex_>& output,
247 const std::optional<std::vector<InputIndex_> >& nnz_consistent,
248 const int threads
249) {
250 // Unlike the corresponding *compressed_sparse() function, we don't try to allocate a big block per thread and fill it up.
251 // We don't have a concept of a Pointer_ type here and we can't be sure that the number of non-zeros wouldn't overflow whatever type we picked internally
252 // (or indeed, the maximum size of a std::vector, for that matter).
253 // So, we just allocate one vector per secondary dimension element, even though that does have some higher overhead.
254 const bool do_parallel = threads > 0;
255 InputIndex_ last_secondary_added = 0;
256 std::optional<std::vector<std::vector<StoredValue_> > > all_partial_values;
257 std::optional<std::vector<std::vector<StoredIndex_> > > all_partial_primary_indices;
258 if (do_parallel) {
259 all_partial_values.emplace(sanisizer::cast<I<decltype(all_partial_values->size())> >(secondary));
260 all_partial_primary_indices.emplace(sanisizer::cast<I<decltype(all_partial_primary_indices->size())> >(secondary));
261 }
262
263 const bool is_sparse = matrix.is_sparse();
264 parallelize([&](const int, const InputIndex_ start, const InputIndex_ length) -> void {
265 if (start == 0) {
266 last_secondary_added = start + length;
267
268 if (is_sparse){
269 Options opt;
270 opt.sparse_ordered_index = false;
271 auto wrk = consecutive_extractor<true>(matrix, !row, start, length, opt);
272 auto buffer_v = create_container_of_Index_size<std::vector<InputValue_> >(primary);
273 auto buffer_i = create_container_of_Index_size<std::vector<InputIndex_> >(primary);
274 for (InputIndex_ x = 0; x < length; ++x) {
275 const auto range = wrk->fetch(buffer_v.data(), buffer_i.data());
276 for (InputIndex_ i = 0; i < range.number; ++i) {
277 output.value[range.index[i]].push_back(range.value[i]);
278 output.index[range.index[i]].push_back(x); // start == 0, so no need to add it to 'x'.
279 }
280 }
281
282 } else {
283 auto wrk = consecutive_extractor<false>(matrix, !row, start, length);
284 auto buffer_v = create_container_of_Index_size<std::vector<InputValue_> >(primary);
285 for (InputIndex_ x = 0; x < length; ++x) {
286 const auto ptr = wrk->fetch(buffer_v.data());
287 for (InputIndex_ p = 0; p < primary; ++p) {
288 const auto val = ptr[p];
289 if (val != 0) {
290 output.value[p].push_back(val);
291 output.index[p].push_back(x); // start == 0, so no need to add it to 'x'.
292 }
293 }
294 }
295 }
296
297 } else {
298 if (is_sparse){
299 Options opt;
300 opt.sparse_ordered_index = false;
301 auto wrk = consecutive_extractor<true>(matrix, !row, start, length, opt);
302 auto buffer_v = create_container_of_Index_size<std::vector<InputValue_> >(primary);
303 auto buffer_i = create_container_of_Index_size<std::vector<InputIndex_> >(primary);
304 for (InputIndex_ x = 0; x < length; ++x) {
305 const auto range = wrk->fetch(buffer_v.data(), buffer_i.data());
306 (*all_partial_values)[start + x] = std::vector<StoredValue_>(range.value, range.value + range.number);
307 (*all_partial_primary_indices)[start + x] = std::vector<StoredIndex_>(range.index, range.index + range.number);
308 }
309
310 } else {
311 auto wrk = consecutive_extractor<false>(matrix, !row, start, length);
312 auto buffer_v = create_container_of_Index_size<std::vector<InputValue_> >(primary);
313
314 for (InputIndex_ x = 0; x < length; ++x) {
315 const auto nnz = (*nnz_consistent)[start + x];
316 std::vector<StoredValue_> out_v;
317 std::vector<StoredIndex_> out_i;
318 out_v.reserve(nnz);
319 out_i.reserve(nnz);
320
321 const auto ptr = wrk->fetch(buffer_v.data());
322 for (InputIndex_ p = 0; p < primary; ++p) {
323 const auto val = ptr[p];
324 if (val != 0) {
325 out_v.push_back(val);
326 out_i.push_back(p);
327 }
328 }
329
330 (*all_partial_values)[start + x] = std::move(out_v);
331 (*all_partial_primary_indices)[start + x] = std::move(out_i);
332 }
333 }
334 }
335 }, secondary, threads);
336
337 if (do_parallel) {
338 for (InputIndex_ s = last_secondary_added; s < secondary; ++s) {
339 const auto& cur_primary_indices = (*all_partial_primary_indices)[s];
340 const auto& cur_values = (*all_partial_values)[s];
341 const auto cur_count = cur_values.size();
342 for (I<decltype(cur_count)> i = 0; i < cur_count; ++i) {
343 const auto primary = cur_primary_indices[i];
344 output.value[primary].push_back(cur_values[i]);
345 output.index[primary].push_back(s);
346 }
347 }

366template<typename StoredValue_, typename StoredIndex_, typename InputValue_, typename InputIndex_>
367FragmentedSparseContents<StoredValue_, StoredIndex_> retrieve_fragmented_sparse_contents(
368 const Matrix<InputValue_, InputIndex_>& matrix,
369 const bool row,
370 const RetrieveFragmentedSparseContentsOptions& options
371) {
372 if (row == matrix.prefer_rows()) {
373 return retrieve_fragmented_sparse_contents_consistent<StoredValue_, StoredIndex_>(matrix, row, options);
374 }
375
376 const InputIndex_ NR = matrix.nrow();
377 const InputIndex_ NC = matrix.ncol();
378 const InputIndex_ primary = (row ? NR : NC);
379 const InputIndex_ secondary = (row ? NC : NR);
380
381 if (!options.two_pass) {
382 // In the one-pass strategy, we load everything in a nice format first, then we transpose it in serial.
383 // This avoids messy reallocations when trying to expand vectors on an inconsistent dimension.
384 auto tmp = retrieve_fragmented_sparse_contents_consistent<StoredValue_, StoredIndex_>(matrix, !row, options);
385 auto primary_counts = create_container_of_Index_size<std::vector<InputIndex_> >(primary);
386 for (I<decltype(secondary)> s = 0; s < secondary; ++s) {
387 const auto& sec_indices = tmp.index[s];
388 const auto num = sec_indices.size();
389 for (I<decltype(num)> n = 0; n < num; ++n) {
390 primary_counts[sec_indices[n]] += 1; // addition must be space, this cannot exceed dimension extents.
391 }
392 }
393
394 FragmentedSparseContents<StoredValue_, StoredIndex_> output(primary);
395 for (InputIndex_ p = 0; p < primary; ++p) {
396 output.index[p].reserve(primary_counts[p]);
397 output.value[p].reserve(primary_counts[p]);
398 }
399
400 for (I<decltype(secondary)> s = 0; s < secondary; ++s) {
401 const auto& sec_values = tmp.value[s];
402 const auto& sec_indices = tmp.index[s];
403 const auto num = sec_indices.size();
404 for (I<decltype(num)> n = 0; n < num; ++n) {
405 const auto curp = sec_indices[n];
406 output.value[curp].push_back(sec_values[n]);
407 output.index[curp].push_back(s);
408 }
409 }
410
411 return output;
412 }
413
414 // In the two-pass strategy, we count the number of non-zeros first, then we fill it up in the second pass.
415 auto nnz_counts = count_fragmented_sparse_non_zeros_inconsistent(matrix, primary, secondary, row, options.num_threads);
416 FragmentedSparseContents<StoredValue_, StoredIndex_> output(primary);
417 for (InputIndex_ p = 0; p < primary; ++p) {
418 output.index[p].reserve(nnz_counts.inconsistent[p]);
419 output.value[p].reserve(nnz_counts.inconsistent[p]);
420 }
421 fill_fragmented_sparse_matrix_inconsistent(matrix, primary, secondary, row, output, nnz_counts.consistent, options.num_threads);
422 return output;
423}
