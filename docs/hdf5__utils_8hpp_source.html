62template<bool integer_only, class GroupLike>
63H5::DataSet open_and_check_dataset(const GroupLike& handle, const std::string& name) {
64 // Avoid throwing H5 exceptions.
65 if (!H5Lexists(handle.getId(), name.c_str(), H5P_DEFAULT) || handle.childObjType(name) != H5O_TYPE_DATASET) {
66 throw std::runtime_error("no child dataset named '" + name + "'");
67 }
68
69 auto dhandle = handle.openDataSet(name);
70 auto type = dhandle.getTypeClass();
71 if constexpr(integer_only) {
72 if (type != H5T_INTEGER) {
73 throw std::runtime_error(std::string("expected integer values in the '") + name + "' dataset");
74 }
75 } else {
76 if (type != H5T_INTEGER && type != H5T_FLOAT) {
77 throw std::runtime_error(std::string("expected numeric values in the '") + name + "' dataset");
78 }
79 }
80
81 return dhandle;
82}

84template<int N>
85std::array<hsize_t, N> get_array_dimensions(const H5::DataSet& handle, const std::string& name) {
86 auto space = handle.getSpace();
87
88 int ndim = space.getSimpleExtentNdims();
89 if (ndim != N) {
90 throw std::runtime_error(std::string("'") + name + "' should be a " + std::to_string(N) + "-dimensional array");
91 }
92
93 std::array<hsize_t, N> dims_out;
94 space.getSimpleExtentDims(dims_out.data(), NULL);
95 return dims_out;
96}
