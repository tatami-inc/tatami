101 void add(const B* buffer, size_t n, Store& store) {
102 size_t i = 0;
103 while (i < n) {
104 if (buffer[i] == '%') {
105 in_comment = true;
106
107 // Try to get quickly to the next line.
108 do {
109 ++i;
110 } while (i < n && buffer[i] != '\n');
111 } else if (buffer[i] == '\n') {
112 new_line(store);
113 ++i;
114 } else if (in_comment) {
115 // Try to get to the next line, again.
116 do {
117 ++i;
118 } while (i < n && buffer[i] != '\n');
119 } else {
120 while (i < n) {
121 if (!std::isdigit(buffer[i])) {
122 if (!std::isspace(buffer[i])) {
123 throw std::runtime_error("values should be non-negative integers on line " + std::to_string(current_line));
124 }
125
126 if (buffer[i] == '\n') {
127 break;
128 }
129
130 if (non_empty) {
131 ++onto;
132 non_empty = false;
133 }
134 ++i;
135 continue;
136 }
137
138 non_empty = true; // check that something was _actually_ present in this field.
139 int delta = buffer[i] - '0';
140 switch(onto) {
141 case 0:
142 currow *= 10;
143 currow += delta;
144 break;
145 case 1:
146 curcol *= 10;
147 curcol += delta;
148 break;
149 case 2:
150 curval *= 10;
151 curval += delta;
152 break;
153 }
154 ++i;
155 }
156 }
157 }
158 return;

161 template<class Store>
162 void finish(Store& store) {
163 // If onto = 0 and non_empty = false, we ended on a newline, so
164 // there's no extra entry to add. Otherwise, we try to add the
165 // last line that was _not_ terminated by a newline.
166 if (onto != 0 || non_empty) {
