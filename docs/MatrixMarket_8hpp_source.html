63 template<class Store>
64 void new_line(Store& store) {
65 if (in_comment) {
66 in_comment = false;
67 } else {
68 if ((onto == 3 && !non_empty) || (onto == 2 && non_empty)) {
69 // i.e., there are three fields.
70 ;
71 } else {
72 throw std::runtime_error("line " + std::to_string(current_line + 1) + " should contain three values");
73 }
74
75 if (!passed_preamble) {
76 nrows = currow;
77 ncols = curcol;
78 nlines = curval;
79
80 store.setdim(currow, curcol, curval);
81 passed_preamble = true;
82 } else {
83 if (!currow) {
84 throw std::runtime_error("row index must be positive on line " + std::to_string(current_line + 1));
85 }
86 if (currow > nrows) {
87 throw std::runtime_error("row index out of range on line " + std::to_string(current_line + 1));
88 }
89
90 if (!curcol) {
91 throw std::runtime_error("column index must be positive on line " + std::to_string(current_line + 1));
92 }
93 if (curcol > ncols) {
94 throw std::runtime_error("column index out of range on line " + std::to_string(current_line + 1));
95 }
96
97 if (current_data_line >= nlines) {
98 throw std::runtime_error("more lines present than specified in the header (" + std::to_string(nlines) + ")");
99 }
100
101 if constexpr(!preamble_only<Store>::value) {
102 store.addline(currow - 1, curcol - 1, curval, current_data_line);
103 }
104 ++current_data_line;
105 }
106
107 onto = 0;
108 currow = 0;
109 curcol = 0;
110 curval = 0;
111 non_empty = false;
112 }
113 ++current_line;
114 }

117 template<class Reader, class Store>
118 void operator()(Reader& reader, Store& store) {
119 bool remaining;
120
121 do {
122 remaining = reader();
123 auto buffer = reinterpret_cast<const char*>(reader.buffer());
124 auto n = reader.available();
125
126 size_t i = 0;
127 while (i < n) {
128 if (buffer[i] == '%') {
129 in_comment = true;
130
131 // Try to get quickly to the next line.
132 do {
133 ++i;
134 } while (i < n && buffer[i] != '\n');
135 } else if (buffer[i] == '\n') {
136 new_line(store);
137 ++i;
138
139 // Breaking out early if we only care about the preamble.
140 if constexpr(preamble_only<Store>::value) {
141 if (passed_preamble) {
142 return;
143 }
144 }
145 } else if (in_comment) {
146 // Try to get to the next line, again.
147 do {
148 ++i;
149 } while (i < n && buffer[i] != '\n');
150 } else {
151 while (i < n) {
152 if (!std::isdigit(buffer[i])) {
153 if (!std::isspace(buffer[i])) {
154 throw std::runtime_error("values should be non-negative integers on line " + std::to_string(current_line));
155 }
156
157 if (buffer[i] == '\n') {
158 break;
159 }
160
161 if (non_empty) {
162 ++onto;
163 non_empty = false;
164 }
165 ++i;
166 continue;
167 }
168
169 non_empty = true; // check that something was _actually_ present in this field.
170 int delta = buffer[i] - '0';
171 switch(onto) {
172 case 0:
173 currow *= 10;
174 currow += delta;
175 break;
176 case 1:
177 curcol *= 10;
178 curcol += delta;
179 break;
180 case 2:
181 curval *= 10;
182 curval += delta;
183 break;
184 }
185 ++i;
186 }
187 }
188 }
189 } while (remaining);
190
191 // If onto = 0 and non_empty = false, we ended on a newline, so
192 // there's no extra entry to add. Otherwise, we try to add the
193 // last line that was _not_ terminated by a newline.
194 if (onto != 0 || non_empty) {
195 new_line(store);
196 }
197
198 if (!passed_preamble) {
199 throw std::runtime_error("no header line specifying the dimensions");
200 }
201
202 if (current_data_line != nlines) {
203 throw std::runtime_error("detected " + std::to_string(current_data_line) + " lines but " + std::to_string(nlines) + " lines specified in the header");
204 }
205 return;
206 }
