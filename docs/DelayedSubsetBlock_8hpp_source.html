26template<typename Index_>
27void bump_indices(VectorPtr<Index_>& indices_ptr, const Index_ subset_start) {
28 if (subset_start) {
29 const auto ptr2 = new std::vector<Index_>(*indices_ptr);
30 indices_ptr.reset(ptr2);
31 for (auto& i : *ptr2) {
32 i += subset_start;
33 }
34 }
