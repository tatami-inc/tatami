270 const Value_* fetch_internal(Index_& i, Value_* buffer) {
271 auto vbuffer = internal_vbuffer.data();
272 auto range = this->internal->fetch(i, vbuffer, internal_ibuffer.data());
273 if (range.value != vbuffer) {
274 // Don't use fetch_copy as we don't want to copy the indices.
275 std::copy(range.value, range.value + range.number, vbuffer);
276 }
277
278 this->parent->operation.template sparse<accrow_>(i, range.number, vbuffer, range.index);
279
280 auto full_length = extracted_length<selection_, Index_>(*(this->internal));
281 if (range.number < full_length) { // avoid calling zero() if possible, as this might throw zero-related errors in non-IEEE platforms.
282 std::fill(buffer, buffer + full_length, [&]{
283 if constexpr(Operation_::always_sparse) {
284 return static_cast<Value_>(0);
285 } else if constexpr(Operation_::always_dense) {
286 return this->parent->operation.template zero<accrow_>(i);
287 } else if (this->parent->operation.actual_sparse()) {
288 return static_cast<Value_>(0);
289 } else {
290 return this->parent->operation.template zero<accrow_>(i);
291 }
292 }());
293 }
294
295 if constexpr(selection_ == DimensionSelectionType::FULL) {
296 for (Index_ i = 0; i < range.number; ++i) {
297 buffer[range.index[i]] = vbuffer[i];
298 }
299 } else if constexpr(selection_ == DimensionSelectionType::BLOCK) {
300 auto shift = this->internal->block_start;
301 for (Index_ i = 0; i < range.number; ++i) {
302 buffer[range.index[i] - shift] = vbuffer[i];
303 }
304 } else {
305 for (Index_ i = 0; i < range.number; ++i) {
306 buffer[index_mapping[range.index[i]]] = vbuffer[i];
307 }
308 }
309
310 return buffer;

555 // Same logic as in SparseIsometricExtractor_NeedsIndices.
556 Index_* iin = (internal_ibuffer.empty() ? ibuffer : internal_ibuffer.data());

558 auto tmp_vbuffer = internal_vbuffer.data();

560 SparseRange<Value_, Index_> output(extracted_length<selection_, Index_>(*this), NULL, NULL);
561
562 if (range.value) {

564 // Don't use fetch_copy as we don't want to copy the indices.

567 this->parent->operation.template sparse<accrow_>(i, range.number, tmp_vbuffer, range.index);
568

570 if (range.number < N) { // only invoking 'zero()' if we really need to, as this could throw various zero-related errors.

620 template<bool accrow_, bool oracle_, DimensionSelectionType selection_, bool sparse_, typename ... Args_>
621 std::unique_ptr<GeneralizedExtractor<oracle_, selection_, sparse_, Value_, Index_> > propagate(const Options& opt, Args_&& ... args) const {
622 std::unique_ptr<GeneralizedExtractor<oracle_, selection_, sparse_, Value_, Index_> > output;
623
624 if constexpr(!sparse_) {
625 if (!(mat->sparse())) {
626 output.reset(new DenseIsometricExtractor_Basic<accrow_, oracle_, selection_>(this, opt, std::forward<Args_>(args)...));
627
628 } else {
629 if (preserves_sparsity()) {
630 output.reset(new DenseIsometricExtractor_FromSparse<accrow_, oracle_, selection_>(this, opt, std::forward<Args_>(args)...));
631
632 } else {
633 if constexpr(accrow_ && !Operation_::needs_column) { // constant sparsity breaking
634 output.reset(new DenseIsometricExtractor_FromSparse<accrow_, oracle_, selection_>(this, opt, std::forward<Args_>(args)...));
635 } else if constexpr(!accrow_ && !Operation_::needs_row) { // constant sparsity breaking
636 output.reset(new DenseIsometricExtractor_FromSparse<accrow_, oracle_, selection_>(this, opt, std::forward<Args_>(args)...));
637 } else { // variable sparsity breaking
638 output.reset(new DenseIsometricExtractor_Basic<accrow_, oracle_, selection_>(this, opt, std::forward<Args_>(args)...));
639 }
640 }
641 }
642
643 } else {
644 if (!(mat->sparse())) {
645 output.reset(new SparseIsometricExtractor_FromDense<accrow_, oracle_, selection_>(this, opt, std::forward<Args_>(args)...));
646
647 } else {
648 if (preserves_sparsity()) {
649 if constexpr(accrow_ && !Operation_::needs_column) { // no need for column indices
650 output.reset(new SparseIsometricExtractor_Simple<accrow_, oracle_, selection_>(this, opt, std::forward<Args_>(args)...));
651 } else if constexpr(!accrow_ && !Operation_::needs_row) { // no need for row indices
652 output.reset(new SparseIsometricExtractor_Simple<accrow_, oracle_, selection_>(this, opt, std::forward<Args_>(args)...));
653 } else { // guess we need indices
654 output.reset(new SparseIsometricExtractor_NeedsIndices<accrow_, oracle_, selection_>(this, opt, std::forward<Args_>(args)...));
655 }
656
657 } else {
658 if constexpr(accrow_ && !Operation_::needs_column) { // constant sparsity breaking
659 output.reset(new SparseIsometricExtractor_ForcedDense<accrow_, oracle_, selection_>(this, opt, std::forward<Args_>(args)...));
660 } else if constexpr(!accrow_ && !Operation_::needs_row) { // constant sparsity breaking
661 output.reset(new SparseIsometricExtractor_ForcedDense<accrow_, oracle_, selection_>(this, opt, std::forward<Args_>(args)...));
662 } else { // variable sparsity breaking
663 output.reset(new SparseIsometricExtractor_FromDense<accrow_, oracle_, selection_>(this, opt, std::forward<Args_>(args)...));
664 }
665 }
666 }
667 }
668
669 return output;
670 }
