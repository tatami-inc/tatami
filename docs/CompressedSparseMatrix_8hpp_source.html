39
40 const Value_* fetch(Index_ i, Value_* buffer) {

429 CompressedSparseMatrix(Index_ nr, Index_ nc, ValueStorage_ vals, IndexStorage_ idx, PointerStorage_ ptr, bool row, bool check=true) :
430 nrows(nr), ncols(nc), values(std::move(vals)), indices(std::move(idx)), indptrs(std::move(ptr)), csr(row)
431 {
432 if (check) {
433 if (values.size() != indices.size()) {
434 throw std::runtime_error("'values' and 'indices' should be of the same length");
435 }
436
437 if (csr) {
438 if (indptrs.size() != static_cast<size_t>(nrows) + 1){
439 throw std::runtime_error("length of 'indptrs' should be equal to 'nrows + 1'");
440 }
441 } else {
442 if (indptrs.size() != static_cast<size_t>(ncols) + 1){
443 throw std::runtime_error("length of 'indptrs' should be equal to 'ncols + 1'");
444 }
445 }
446
447 if (indptrs[0] != 0) {
448 throw std::runtime_error("first element of 'indptrs' should be zero");
449 }
450
451 auto last = indptrs[indptrs.size() - 1]; // don't use back() as this is not guaranteed to be available for arbitrary PointerStorage_.
452 if (static_cast<size_t>(last) != indices.size()) {
453 throw std::runtime_error("last element of 'indptrs' should be equal to length of 'indices'");
454 }
455
456 ElementType<IndexStorage_> max_index = (csr ? ncols : nrows);
457 for (size_t i = 1; i < indptrs.size(); ++i) {
458 auto start = indptrs[i- 1], end = indptrs[i];
459 if (end < start || end > last) {
460 throw std::runtime_error("'indptrs' should be in non-decreasing order");
461 }
462
463 for (auto x = start; x < end; ++x) {
464 if (indices[x] < 0 || indices[x] >= max_index) {
465 throw std::runtime_error("'indices' should contain non-negative integers less than the number of " + (csr ? std::string("columns") : std::string("rows")));
466 }
467 }
468
469 for (size_t j = start + 1; j < end; ++j) {
470 if (indices[j] <= indices[j - 1]) {
471 throw std::runtime_error("'indices' should be strictly increasing within each " + (csr ? std::string("row") : std::string("column")));
472 }
473 }
474 }
475 }
