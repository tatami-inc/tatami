32void convert_to_dense(const Matrix<InputValue_, InputIndex_>* incoming, StoredValue_* store, int threads = 1) {
33 InputIndex_ NR = incoming->nrow();
34 InputIndex_ NC = incoming->ncol();
35 InputIndex_ primary = (row_ ? NR : NC);
36 InputIndex_ secondary = (row_ ? NC : NR);
37
38 if (row_ == incoming->prefer_rows()) {
39 constexpr bool same_type = std::is_same<InputValue_, StoredValue_>::value;
40 parallelize([&](size_t, InputIndex_ start, InputIndex_ length) -> void {
41 std::vector<InputValue_> temp(same_type ? 0 : secondary);
42 auto store_copy = store + start * secondary;
43 auto wrk = consecutive_extractor<row_, false>(incoming, start, length);
44
45 for (InputIndex_ p = start, e = start + length; p < e; ++p, store_copy += secondary) {
46 if constexpr(same_type) {
47 wrk->fetch_copy(p, store_copy);
48 } else {
49 auto ptr = wrk->fetch(p, temp.data());
50 std::copy(ptr, ptr + secondary, store_copy);
51 }
52 }
53 }, primary, threads);
54
55 } else {
56 // We iterate on the incoming matrix's preferred dimension, under the
57 // assumption that it may be arbitrarily costly to extract in the
58 // non-preferred dim; it is thus cheaper to do cache-unfriendly inserts
59 // into the output buffers.
60
61 parallelize([&](size_t, InputIndex_ start, InputIndex_ length) -> void {
62 auto wrk = consecutive_extractor<!row_, false>(incoming, 0, secondary, start, length);
63 auto len = wrk->block_length;
64 std::vector<InputValue_> temp(len);
65 auto store_copy = store + start * secondary;
66
67 for (InputIndex_ s = 0; s < secondary; ++s, ++store_copy) {
68 auto ptr = wrk->fetch(s, temp.data());
69 auto bptr = store_copy;
70 for (InputIndex_ p = 0; p < len; ++p, bptr += secondary) {
71 *bptr = ptr[p];
72 }
73 }
74 }, primary, threads);
75 }
76
77 return;
78}
