20LayeredMatrixData<T, IDX> convert_to_layered_sparse_internal(const Matrix<T, IDX>* incoming) {
21 auto ranges = row_ranges(incoming);
22 const auto& mins = ranges.first;
23 const auto& maxs = ranges.second;
24
25 // Using the permutation vector to store the reindexing vector for the time being.
26 LayeredMatrixData<T, IDX> output;
27 auto& reindexed = output.permutation;
28 reindexed.resize(maxs.size());
29
30 // Choosing a category for each row.
31 std::vector<uint8_t> category(maxs.size());
32 constexpr double max8 = std::numeric_limits<uint8_t>::max();
33 constexpr double max16 = std::numeric_limits<uint16_t>::max();
34 std::vector<size_t> per_category(3);
35
36 for (size_t i = 0; i < maxs.size(); ++i) {
37 if (mins[i] < 0) {
38 throw std::runtime_error("all values must be non-negative");
39 }
40 if (maxs[i] <= max8) {
41 category[i] = 0;
42 } else if (maxs[i] <= max16) {
43 category[i] = 1;
44 } else {
45 category[i] = 2;
46 }
47
48 auto& sofar = per_category[category[i]];
49 reindexed[i] = sofar;
50 ++sofar;
51 }
52
53 // Filling the categories.
54 std::vector<ROW> row8;
55 std::vector<uint32_t> col8;
56 std::vector<uint8_t> dat8;
57
58 std::vector<ROW> row16;
59 std::vector<uint32_t> col16;
60 std::vector<uint16_t> dat16;
61
62 std::vector<ROW> row32;
63 std::vector<uint32_t> col32;
64 std::vector<uint32_t> dat32;
65
66 auto add_nonzero_element = [&](size_t r, size_t c, T val) -> void {
67 auto cat = category[r];
68 auto r2 = reindexed[r];
69 switch (cat) {
70 case 0:
71 row8.push_back(r2);
72 col8.push_back(c);
73 dat8.push_back(val);
74 break;
75 case 1:
76 row16.push_back(r2);
77 col16.push_back(c);
78 dat16.push_back(val);
79 break;
80 case 2:
81 row32.push_back(r2);
82 col32.push_back(c);
83 dat32.push_back(val);
84 break;
85 }
86 };
87
88 if (incoming->sparse()) {
89 if (incoming->prefer_rows()) {
90 auto wrk = incoming->new_workspace(true);
91 std::vector<IDX> ibuffer(incoming->ncol());
92 std::vector<T> vbuffer(incoming->ncol());
93
94 for (size_t r = 0; r < incoming->nrow(); ++r) {
95 auto range = incoming->sparse_row(r, vbuffer.data(), ibuffer.data(), wrk.get());
96 for (size_t i = 0; i < range.number; ++i) {
97 if (range.value[i]) {
98 add_nonzero_element(r, range.index[i], range.value[i]);
99 }
100 }
101 }
102
103 } else {
104 auto wrk = incoming->new_workspace(false);
105 std::vector<IDX> ibuffer(incoming->nrow());
106 std::vector<T> vbuffer(incoming->nrow());
107

110 for (size_t i = 0; i < range.number; ++i) {
111 if (range.value[i]) {

113 }
114 }
115 }
116 }
117
118 } else {
119 if (incoming->prefer_rows()) {
120 auto wrk = incoming->new_workspace(true);
121 std::vector<T> buffer(incoming->ncol());
122
123 for (size_t r = 0; r < incoming->nrow(); ++r) {
124 auto ptr = incoming->row(r, buffer.data(), wrk.get());
125 for (size_t c = 0; c < incoming->ncol(); ++c) {
126 if (ptr[c] != 0) {
127 add_nonzero_element(r, c, ptr[c]);
128 }
129 }
130 }
131
132 } else {
133 auto wrk = incoming->new_workspace(false);
134 std::vector<T> buffer(incoming->nrow());
135
136 for (size_t c = 0; c < incoming->ncol(); ++c) {
137 auto ptr = incoming->column(c, buffer.data(), wrk.get());
138 for (size_t r = 0; r < incoming->nrow(); ++r) {
139 if (ptr[r] != 0) {
140 add_nonzero_element(r, c, ptr[r]);
141 }
142 }
143 }
144 }
145 }

147 // Creating the matrix.
148 std::vector<std::shared_ptr<Matrix<T, IDX> > > collated;
149 size_t NC = incoming->ncol();
150
151 auto create_sparse_matrix = [](size_t nr, size_t nc, auto values, auto rows, auto cols) -> auto {
152 auto indptrs = compress_sparse_triplets<false>(nr, nc, values, rows, cols);
153 typedef CompressedSparseColumnMatrix<T, IDX, decltype(values), decltype(rows), decltype(indptrs)> CSCMatrix;
154 return std::shared_ptr<Matrix<T, IDX> >(new CSCMatrix(nr, nc, std::move(values), std::move(rows), std::move(indptrs)));
155 };
156
157 if (per_category[0]) {
158 collated.push_back(create_sparse_matrix(per_category[0], NC, std::move(dat8), std::move(row8), std::move(col8)));
159 }
160 if (per_category[1]) {
161 collated.push_back(create_sparse_matrix(per_category[1], NC, std::move(dat16), std::move(row16), std::move(col16)));
162 }
163 if (per_category[2]) {
164 collated.push_back(create_sparse_matrix(per_category[2], NC, std::move(dat32), std::move(row32), std::move(col32)));
165 }
166
167 if (collated.size() == 0) {
168 output.matrix = create_sparse_matrix(0, NC, std::move(dat8), std::move(row8), std::move(col8));
169 } else if (collated.size() == 1) {
170 output.matrix = collated[0];
171 } else {
172 output.matrix = make_DelayedBind<0>(std::move(collated));
173 }
174
175 // Converting the reindexing vector into the permutation vector.
176 {
177 std::vector<size_t> offset(3);
178 offset[1] = per_category[0];
179 offset[2] = per_category[0] + per_category[1];
180 for (size_t i = 0 ; i < category.size(); ++i) {
181 output.permutation[i] += offset[category[i]];
182 }
183 }
184
185 return output;
186}

210 if (incoming->nrow() <= max16) {
